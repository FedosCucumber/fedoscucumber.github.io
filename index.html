        public class Solution {
    public int Search(int[] nums, int target) {
        int left = 0;
        int right = nums.Length;
        int pivot = (left + right) / 2;
        while (left + 1 < right || nums[pivot] == target)
        {
            if (nums[pivot] == target)
            {
                return pivot;
            }
            if (nums[pivot] < target)
            {
                left = pivot;
            }
            else
            {
                right = pivot;
            }

            pivot = (left + right) / 2;
        }

        return -1;
    }
}
